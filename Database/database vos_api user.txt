-- DROP ROLE vos_api;

CREATE ROLE vos_api WITH 
	NOSUPERUSER
	NOCREATEDB
	NOCREATEROLE
	NOINHERIT
	LOGIN
	NOREPLICATION
	NOBYPASSRLS
	CONNECTION LIMIT UNLIMITED;

-- Permissions

GRANT INSERT, SELECT, UPDATE, DELETE ON TABLE public.admission_types TO vos_api;
GRANT INSERT, SELECT, UPDATE, DELETE ON TABLE public.allocation_category TO vos_api;
GRANT INSERT, SELECT, UPDATE, DELETE ON TABLE public.application TO vos_api;
GRANT INSERT, SELECT, UPDATE, DELETE ON TABLE public.board_type TO vos_api;
GRANT INSERT, SELECT, UPDATE, DELETE ON TABLE public.branch TO vos_api;
GRANT INSERT, SELECT, UPDATE, DELETE ON TABLE public.guardian TO vos_api;
GRANT INSERT, SELECT, UPDATE, DELETE ON TABLE public.parent TO vos_api;
GRANT USAGE, CREATE ON SCHEMA public TO vos_api;
GRANT INSERT, SELECT, UPDATE, DELETE ON TABLE public.student TO vos_api;
